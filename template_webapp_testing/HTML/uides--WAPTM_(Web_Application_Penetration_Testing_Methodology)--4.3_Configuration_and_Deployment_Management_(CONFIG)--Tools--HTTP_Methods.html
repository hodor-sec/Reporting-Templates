HTTP Methods


Loop through HTTP methods
host=www.sec542.org; for method in GET POST PUT TRACE CONNECT OPTIONS; do printf "$host: $method\n\n"; printf "$method / HTTP/1.1\r\nHost: $host\r\n\r\n" | nc $host 80; printf '\n'; done

Loop through HTTP methods, HTTPS
host=www.sec542.org; for method in GET POST PUT TRACE CONNECT OPTIONS; do printf "$host: $method\n\n"; printf "$method / HTTP/1.1\r\nHost: $host\r\n\r\n" | ncat --ssl $host 443; printf '\n'; done

CURL method HTTP
host=www.sec542.org; for method in GET POST PUT TRACE CONNECT OPTIONS; do printf "$host: $method\n\n"; curl -I -X $method http://$host; done

CURL method HTTPS
host=www.sec542.org; for method in GET POST PUT TRACE CONNECT OPTIONS; do printf "$host: $method\n\n"; curl -kI -X $method https://$host; done
