Version syntax


HTTP 0.9
$ echo -e "GET / HTTP/0.9\r\n" | nc localhost 80

HTTP 1.0
$ echo -e "GET / HTTP/1.0\r\n" | nc localhost 80

HTTP 1.1
$ echo -e "GET / HTTP/1.1\r\nHost: localhost:80\r\n\r\n" | nc localhost 80
$ echo -e "GET / HTTP/1.1\r\nHost: localhost:80\r\nConnection: close\r\n\r\n\r\n" | nc localhost 80
