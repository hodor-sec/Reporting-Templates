"""Generate a base64 encoded HMAC signature.

"""

__module_id__ = "$Id$"
#end_pymotw_header

import base64
import hmac
import hashlib

with open('lorem.txt', 'rb') as f:
    body = f.read()

hash = hmac.new('secret-shared-key-goes-here', body, hashlib.sha1)
digest = hash.digest()
print base64.encodestring(digest)
